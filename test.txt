int jopa(int n, int prime[0])
{
int ind = 2;
while(ind < n+1){
	prime[ind] = 1;
	ind = ind + 1;
}
int j;
ind = 2;
while(ind<=n){
	if(prime[ind]){
		if(ind*ind <= n){
			j = ind*ind;
			while(j<=n){
				prime[j] = 0;
				j = j + ind;
			}
		}
	}
	ind = ind+1;
	
}
return 0;
}

int main(){
	int n;
	cin >> n;
	int prime[n+1];
	<int>jopa(n, prime[0]);
	int ind = 0;
	while(ind <= n){
		if(prime[ind]){
			cout << ind << ' ';
		}
		ind = ind + 1;
	}
	return 0;
}
